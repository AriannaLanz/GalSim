00522     void KTable::fill(const T& f) 
00523     {
00524         clearCache(); // invalidate any stored interpolations
00525         std::complex<double>* zptr=_array.get();
00526         double kx, ky;
00527         for (int iy=0; iy< _No2; iy++) {
00528             ky = iy*_dk;
00529             for (int ix=0; ix< _No2+1 ; ix++) {
00530                 kx = ix*_dk;
00531                 *(zptr++) = f(kx,ky);
00532             }
00533         }
00534         // wrap to the negative ky's
00535         for (int iy=-_No2; iy< 0; iy++) {
00536             ky = iy*_dk;
00537             for (int ix=0; ix< _No2+1 ; ix++) {
00538                 kx = ix*_dk;
00539                 *(zptr++) = f(kx,ky);
00540             }
00541         }
00542         return;
