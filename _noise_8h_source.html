246  template <typename T>
247  void applyTo(ImageView<T> data)
248  {
249  // Above this many e's, assume Poisson distribution == Gaussian
250  // The Gaussian deviate is about 20% faster than Poisson, and for high N
251  // they are virtually identical.
252  const double MAX_POISSON=1.e5;
253  // Typedef for image row iterable
254  typedef typename ImageView<T>::iterator ImIter;
255 
256  data += T(_sky_level);
257 
258  PoissonDeviate pd(*_rng, 1.); // will reset the mean for each pixel below.
259  GaussianDeviate gd(*_rng, 0., 1.);
260  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
261  ImIter ee = data.rowEnd(y);
262  for (ImIter it = data.rowBegin(y); it != ee; ++it) {
263  if (*it <= 0.) continue;
264  if (*it < MAX_POISSON) {
265  pd.setMean(*it);
266  *it = T(pd());
267  } else {
268  gd.setSigma(sqrt(*it));
269  *it = T(*it + gd());
270  }
271  }
272  }

275  }

405  void applyTo(ImageView<T> data)
406  {
407  // Above this many e's, assume Poisson distribution == Gaussian
408  // The Gaussian deviate is about 20% faster than Poisson, and for high N
409  // they are virtually identical.
410  const double MAX_POISSON=1.e5;
411  // Typedef for image row iterable
412  typedef typename ImageView<T>::iterator ImIter;
413 
414  data += T(_sky_level);
415 
416  // Add the Poisson noise first:
417  if (_gain > 0.) {
418  PoissonDeviate pd(*_rng, 1.); // will reset the mean for each pixel below.
419  GaussianDeviate gd(*_rng, 0., 1.);
420  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
421  ImIter ee = data.rowEnd(y);
422  for (ImIter it = data.rowBegin(y); it != ee; ++it) {
423  if (*it <= 0.) continue;
424  double electrons = *it * _gain;
425  if (electrons < MAX_POISSON) {
426  pd.setMean(electrons);
427  *it = T(pd() / _gain);
428  } else {
429  gd.setSigma(sqrt(electrons)/_gain);
430  *it = T(*it + gd());
431  }
432  }
433  }
434  }

461  void applyToVar(ImageView<T> data, ImageView<T> var)
462  {
463  // Typedef for image row iterable
464  typedef typename ImageView<T>::iterator ImIter;
465  assert(data.getBounds() == var.getBounds());
466  // Fill with the (constant) Gaussian contribution to variance
467  if (_read_noise > 0.) {
468  double sigma = _read_noise / (_gain > 0. ? _gain : 1.);
469  var.fill(sigma * sigma);
470  }
471  // Add the Poisson variance:
472  if (_gain > 0.) {
473  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
474  ImIter ee = data.rowEnd(y);

476  for (ImIter it = data.rowBegin(y); it != ee; ++it, ++it2) {
477  if (*it > 0.) *it2 += (*it + _sky_level) / _gain;
478  }
479  }
