281  template <typename T>
282  void applyToView(ImageView<T> data)
283  {
284  // Above this many e's, assume Poisson distribution == Gaussian
285  // The Gaussian deviate is about 20% faster than Poisson, and for high N
286  // they are virtually identical.
287  const double MAX_POISSON=1.e5;
288  // Typedef for image row iterable
289  typedef typename ImageView<T>::iterator ImIter;
290 
291  data += T(_sky_level);
292 
293  PoissonDeviate pd(*_rng, 1.); // will reset the mean for each pixel below.
294  GaussianDeviate gd(*_rng, 0., 1.);
295  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
296  ImIter ee = data.rowEnd(y);
297  for (ImIter it = data.rowBegin(y); it != ee; ++it) {
298  if (*it <= 0.) continue;
299  if (*it < MAX_POISSON) {
300  pd.setMean(*it);
301  *it = T(pd());
302  } else {
303  gd.setSigma(sqrt(*it));
304  *it = T(*it + gd());
305  }
306  }
307  }
308 
309  data -= T(_sky_level);
310  }

442  template <typename T>
443  void applyToView(ImageView<T> data)
444  {
445  // Above this many e's, assume Poisson distribution == Gaussian
446  // The Gaussian deviate is about 20% faster than Poisson, and for high N
447  // they are virtually identical.
448  const double MAX_POISSON=1.e5;
449  // Typedef for image row iterable
450  typedef typename ImageView<T>::iterator ImIter;
451 
452  data += T(_sky_level);
453 
454  // Add the Poisson noise first:
455  if (_gain > 0.) {
456  PoissonDeviate pd(*_rng, 1.); // will reset the mean for each pixel below.
457  GaussianDeviate gd(*_rng, 0., 1.);
458  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
459  ImIter ee = data.rowEnd(y);
460  for (ImIter it = data.rowBegin(y); it != ee; ++it) {
461  if (*it <= 0.) continue;
462  double electrons = *it * _gain;
463  if (electrons < MAX_POISSON) {
464  pd.setMean(electrons);
465  *it = T(pd() / _gain);
466  } else {
467  gd.setSigma(sqrt(electrons)/_gain);
468  *it = T(*it + gd());
469  }
470  }
471  }
472  }
473 
474  // Next add the Gaussian noise:
475  if (_read_noise > 0.) {
476  GaussianDeviate gd(*_rng, 0., _read_noise / (_gain > 0. ? _gain : 1.));
477  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
478  ImIter ee = data.rowEnd(y);
479  for (ImIter it = data.rowBegin(y); it != ee; ++it) {
480  *it = T(*it + gd());
481  }

500  {
501  // Typedef for image row iterable
502  typedef typename ImageView<T>::iterator ImIter;
503  assert(data.getBounds() == var.getBounds());
504  // Fill with the (constant) Gaussian contribution to variance
505  if (_read_noise > 0.) {
506  double sigma = _read_noise / (_gain > 0. ? _gain : 1.);
507  var.fill(sigma * sigma);
508  }
509  // Add the Poisson variance:
510  if (_gain > 0.) {
511  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y

514  for (ImIter it = data.rowBegin(y); it != ee; ++it, ++it2) {
515  if (*it > 0.) *it2 += (*it + _sky_level) / _gain;
516  }
517  }
518  }
