190  virtual void checkSampler() const
191  {
192  if (_sampler.get()) return;
193  // Will assume by default that the Interpolant kernel changes sign at non-zero
194  // integers, with one extremum in each integer range.
195  int nKnots = int(ceil(xrange()));
196  std::vector<double> ranges(2*nKnots);
197  for (int i=1; i<=nKnots; i++) {
198  double knot = std::min(double(i), xrange());
199  ranges[nKnots-i] = -knot;
200  ranges[nKnots+i-1] = knot;
201  }
202  _sampler.reset(new OneDimensionalDeviate(_interp, ranges, false, _gsparams));
203  }
