157  virtual void checkSampler() const {
158  if (_sampler) return;
159  // Will assume by default that the Interpolant kernel changes sign at non-zero
160  // integers, with one extremum in each integer range.
161  int nKnots = int(ceil(xrange()));
162  std::vector<double> ranges(2*nKnots);
163  for (int i=1; i<=nKnots; i++) {
164  double knot = std::min(1.*i, xrange());
165  ranges[nKnots-i] = -knot;
166  ranges[nKnots+i-1] = knot;
167  }
168  _sampler = new OneDimensionalDeviate(_interp, ranges);
