176  virtual void checkSampler() const
177  {
178  if (_sampler.get()) return;
179  // Will assume by default that the Interpolant kernel changes sign at non-zero
180  // integers, with one extremum in each integer range.
181  int nKnots = int(ceil(xrange()));
182  std::vector<double> ranges(2*nKnots);
183  for (int i=1; i<=nKnots; i++) {
184  double knot = std::min(double(i), xrange());
185  ranges[nKnots-i] = -knot;
186  ranges[nKnots+i-1] = knot;
187  }
188  _sampler.reset(new OneDimensionalDeviate(_interp, ranges));
189  }

318  inline double Si(double x)
319  {
320  double x2=x*x;
321  if(x2>=3.8) {
322  // Use rational approximation from Abramowitz & Stegun
323  // cf. Eqns. 5.2.38, 5.2.39, 5.2.8 - where it says it's good to <1e-6.
324  // ain't this pretty?
325  return (M_PI/2.)*((x>0.)?1.:-1.)
326  - (38.102495+x2*(335.677320+x2*(265.187033+x2*(38.027264+x2))))
327  / (x* (157.105423+x2*(570.236280+x2*(322.624911+x2*(40.021433+x2)))) )*std::cos(x)
328  - (21.821899+x2*(352.018498+x2*(302.757865+x2*(42.242855+x2))))
329  / (x2*(449.690326+x2*(1114.978885+x2*(482.485984+x2*(48.196927+x2)))))*std::sin(x);
330 
331  } else {
332  // x2<3.8: the series expansion is the better approximation, A&S 5.2.14
333  double n1=1.;
334  double n2=1.;
335  double tt=x;
336  double t=0;
337  for(int i=1; i<7; i++) {
338  t += tt/(n1*n2);
339  tt = -tt*x2;
340  n1 = 2.*double(i)+1.;
341  n2*= n1*2.*double(i);
342  }
343  return t;
344  }
