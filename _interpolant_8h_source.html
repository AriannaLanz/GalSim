183  // Allocate photon sampler and do all of its pre-calculations
184  virtual void checkSampler() const
185  {
186  if (_sampler.get()) return;
187  // Will assume by default that the Interpolant kernel changes sign at non-zero
188  // integers, with one extremum in each integer range.
189  int nKnots = int(ceil(xrange()));
190  std::vector<double> ranges(2*nKnots);
191  for (int i=1; i<=nKnots; i++) {
192  double knot = std::min(double(i), xrange());
193  ranges[nKnots-i] = -knot;
194  ranges[nKnots+i-1] = knot;
195  }
196  _sampler.reset(new OneDimensionalDeviate(_interp, ranges, false, _gsparams));
197  }
