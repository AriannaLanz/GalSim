155  virtual void checkSampler() const
156  {
157  if (_sampler.get()) return;
158  // Will assume by default that the Interpolant kernel changes sign at non-zero
159  // integers, with one extremum in each integer range.
160  int nKnots = int(ceil(xrange()));
161  std::vector<double> ranges(2*nKnots);
162  for (int i=1; i<=nKnots; i++) {
163  double knot = std::min(double(i), xrange());
164  ranges[nKnots-i] = -knot;
165  ranges[nKnots+i-1] = knot;
166  }
167  _sampler.reset(new OneDimensionalDeviate(_interp, ranges));
168  }

303  // ain't this pretty?

306  / (x* (157.105423+x2*(570.236280+x2*(322.624911+x2*(40.021433+x2)))) )*std::cos(x)
307  - (21.821899+x2*(352.018498+x2*(302.757865+x2*(42.242855+x2))))
308  / (x2*(449.690326+x2*(1114.978885+x2*(482.485984+x2*(48.196927+x2)))))*std::sin(x);
309 
310  } else {
311  // x2<3.8: the series expansion is the better approximation, A&S 5.2.14
312  double n1=1.;
313  double n2=1.;
314  double tt=x;
315  double t=0;
316  for(int i=1; i<7; i++) {
317  t += tt/(n1*n2);
