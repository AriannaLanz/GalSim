19  // Parameter setting class
20  // Pset is a collection of parameters attached to variables in the code.
21 
22  // Pset class can set any of its variables by reading from an input stream.
23  // Each line of the input stream is assumed to have the following format:
24  // <key> [=] <value> [comment]
25  // key is the name of the parameter
26  // optional = sign follows
27  // value is ASCII representation of the value
28  // comment is ; or # character followed by stuff to ignore.
29  //
30  // Blank lines are ignored, as are lines starting with comment characters.
31  // White space always allowed between elements of a line.
32  // String values can be enclosed in quotes, otherwise everything on
33  // the line up to comment delimiter is assumed to be part of string.
34  // Trailing white space is stripped from non-quoted strings.

243  std::string::const_iterator pl=ps->keyword.begin();
244  while (pr!=rhs.end()) {

254  template <class T>
255  void PsetMem<T>::setValue(const std::string& valstring)
256  {
257  if (valstring.length()==0) {
258  if (f & hasDefault) { setDefault(); return; }
259  else throw PsetNoDefault(keyword);
260  }
261  std::istringstream is(valstring);
262  T thisval;
263  is >> thisval;
264  if (is.fail()) throw PsetFormatError(keyword,valstring);
265  if (f & hasLowerBound) {
266  if ( thisval < lowerBound
267  || ( (f&openLowerBound) && thisval == lowerBound) )
268  throw PsetOutOfBounds(keyword, valstring);
269  }
270  if (f & hasUpperBound) {
271  if ( thisval > upperBound
272  || ( (f&openUpperBound) && thisval == upperBound) )
273  throw PsetOutOfBounds(keyword, valstring);
274  }
275  *valueptr = thisval;
276  }
