00082     inline double getBesselRoot0(int s)
00083     {
00084         if (s <= 0)
00085             throw std::runtime_error("s must be > 0");
00086         if (s <= n_roots_j0) return root_j0[s-1];
00087         else {
00088             // Above this value, the asymptotic formula from Abramowitz and Stegun 9.5.12
00089             // is accurate to better than 1.e-16, so essentially exact.
00090             // b - (m-1)/8b - 4(m-1)(7m-31)/3(8b)^3 - 32(m-1)(83m^2-982m+3779)/15(8b)^5
00091             //   - 64(m-1)(6949m^3-153855m^2+1585743m-6277237)/105(8b)^7
00092             // where m = 4nu^2 and b = (s + nu/2 - 1/4)pi.
00093             //
00094             // For nu = 0, this simplifies considerably to:
00095             // b + 1/8b - 4*31/3(8b)^3 + 32*3779/15(8b)^5 - 64*6277237/105(8b)^7
00096             double b = (s - 0.25) * M_PI;
00097             double temp = 0.125/b;
00098             double inv8bsq = temp*temp;
00099 
00100             // From here on, b will be the running total, not the original b, and temp will be 
00101             // the part of each coefficient that is in common with the next term.
00102             b += temp;
00103 
00104             temp *= (4./3.) * inv8bsq;
00105             b -= 31. * temp;
00106 
00107             temp *= (8./5.) * inv8bsq;
00108             b += 3779. * temp;
00109 
00110             temp *= (2./7.) * inv8bsq;
00111             b -= 6277237. * temp;
00112 
00113             return b;
00114         }
00115     }
