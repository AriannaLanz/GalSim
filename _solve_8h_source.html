166  void bracket()
167  {
168  const double factor=2.0;
169  if (uBound == lBound)
170  throw SolveError("uBound=lBound in bracket()");
171  if (!boundsAreEvaluated) {
172  flower = func(lBound);
173  fupper = func(uBound);
174  boundsAreEvaluated=true;
175  }
176 
177  double delta = uBound-lBound;
178  for (int j=1; j<maxSteps; j++) {
179  if (fupper*flower < 0.0) return;
180  if (std::abs(flower) < std::abs(fupper)) {
181  uBound = lBound;
182  fupper = flower;
183  delta *= factor;
184  lBound -= delta;
185  flower = func(lBound);
186  } else {
187  lBound = uBound;
188  flower = fupper;
189  delta *= factor;
190  uBound += delta;
191  fupper = func(uBound);
192  }
193  }
194  throw SolveError("Too many iterations in bracket()");
195  }

272  T bisect() const
273  {
274  T dx,f,fmid,xmid,rtb;
275 
276  if (!boundsAreEvaluated) {
277  flower=func(lBound);
278  fupper=func(uBound);
279  boundsAreEvaluated = true;
280  }
281  f=flower;
282  fmid=fupper;
283 
284  if (f*fmid >= 0.0)
285  FormatAndThrow<SolveError> () << "Root is not bracketed: " << lBound
286  << " " << uBound;
287  rtb = f < 0.0 ? (dx=uBound-lBound,lBound) : (dx=lBound-uBound,uBound);
288  for (int j=1;j<=maxSteps;j++) {
289  fmid=func(xmid=rtb+(dx *= 0.5));
290  if (fmid <= 0.0) rtb=xmid;
291  if ( (std::abs(dx) < xTolerance) || fmid == 0.0) return rtb;
292  }
293  throw SolveError("Too many bisections");
294  return 0.0;

300  T zbrent() const
301  {
302  T a=lBound, b=uBound, c=uBound;
303  T d=b-a, e=b-a;
304  T min1,min2;
305  if (!boundsAreEvaluated) {
306  flower=func(a);
307  fupper=func(b);
308  boundsAreEvaluated = true;
309  }
310  T fa = flower;
311  T fb = fupper;
312 
313  T p,q,r,s,tol1,xm;
314  if ((fa > 0.0 && fb > 0.0) || (fa < 0.0 && fb < 0.0)) {
315  FormatAndThrow<SolveError> () << "Root is not bracketed: "
316  << lBound << " " << uBound;
317  }
318  T fc=fb;
319  for (int iter=0;iter<=maxSteps;iter++) {
320  if ((fb > 0.0 && fc > 0.0) || (fb < 0.0 && fc < 0.0)) {
321  c=a;
322  fc=fa;
323  e=d=b-a;
324  }
325  if (std::abs(fc) < std::abs(fb)) {
326  a=b;
327  b=c;
328  c=a;
329  fa=fb;
330  fb=fc;
331  fc=fa;
332  }
333  tol1=2.0*std::numeric_limits<T>::epsilon()*std::abs(b)
334  +0.5*xTolerance;
335  xm=0.5*(c-b);
336  if (std::abs(xm) <= tol1 || fb == 0.0) return b;
337  if (std::abs(e) >= tol1 && std::abs(fa) > std::abs(fb)) {
338  s=fb/fa;
339  if (a == c) {
340  p=2.0*xm*s;
341  q=1.0-s;
342  } else {
343  q=fa/fc;
344  r=fb/fc;
345  p=s*(2.0*xm*q*(q-r)-(b-a)*(r-1.0));
346  q=(q-1.0)*(r-1.0)*(s-1.0);
347  }
348  if (p > 0.0) q = -q;
349  p=std::abs(p);
350  min1=3.0*xm*q-std::abs(tol1*q);
351  min2=std::abs(e*q);
352  if (2.0*p < std::min(min1,min2) ) {
353  e=d;
354  d=p/q;
355  } else {
356  d=xm;
357  e=d;
358  }
359  } else {
360  d=xm;
361  e=d;
362  }
363  a=b;
364  fa=fb;
365  if (std::abs(d) > tol1) b += d;
366  else b += (xm>=0. ? std::abs(tol1) : -std::abs(tol1));
367  fb=func(b);
368  }
369  throw SolveError("Maximum number of iterations exceeded in zbrent");
370  }
