53  // Hunt for bracket, geometrically expanding range
54  // This version assumes that the root is to the side of the end point that is
55  // closer to 0. This will be true if the function is monotonic.
56  void bracket()
57  {
58  const double factor=2.0;
59  if (uBound == lBound)
60  throw SolveError("uBound=lBound in bracket()");
61  if (!boundsAreEvaluated) {
62  flower = func(lBound);
63  fupper = func(uBound);
64  boundsAreEvaluated=true;
65  }
66 
67  double delta = uBound-lBound;
68  for (int j=1; j<maxSteps; j++) {
69  if (fupper*flower < 0.0) return;
70  if (std::abs(flower) < std::abs(fupper)) {
71  uBound = lBound;
72  fupper = flower;
73  delta *= factor;
74  lBound -= delta;
75  flower = func(lBound);
76  } else {
77  lBound = uBound;
78  flower = fupper;
79  delta *= factor;
80  uBound += delta;
81  fupper = func(uBound);
82  }
83  }
84  throw SolveError("Too many iterations in bracket()");
85  }

175  T zbrent() const
176  {
177  T a=lBound, b=uBound, c=uBound;
178  T d=b-a, e=b-a;
179  T min1,min2;
180  if (!boundsAreEvaluated) {
181  flower=func(a);
182  fupper=func(b);
183  boundsAreEvaluated = true;
184  }
185  T fa = flower;
186  T fb = fupper;
187 
188  T p,q,r,s,tol1,xm;
189  if ((fa > 0.0 && fb > 0.0) || (fa < 0.0 && fb < 0.0)) {
190  FormatAndThrow<SolveError> () << "Root is not bracketed: "
191  << lBound << " " << uBound;
192  }
193  T fc=fb;
194  for (int iter=0;iter<=maxSteps;iter++) {
195  if ((fb > 0.0 && fc > 0.0) || (fb < 0.0 && fc < 0.0)) {
196  c=a;
197  fc=fa;
198  e=d=b-a;
199  }
200  if (std::abs(fc) < std::abs(fb)) {
201  a=b;
202  b=c;
203  c=a;
204  fa=fb;
205  fb=fc;
206  fc=fa;
207  }
208  tol1=2.0*std::numeric_limits<T>::epsilon()*std::abs(b)
209  +0.5*xTolerance;
210  xm=0.5*(c-b);
211  if (std::abs(xm) <= tol1 || fb == 0.0) return b;
212  if (std::abs(e) >= tol1 && std::abs(fa) > std::abs(fb)) {
213  s=fb/fa;
214  if (a == c) {
215  p=2.0*xm*s;
216  q=1.0-s;
217  } else {
218  q=fa/fc;
219  r=fb/fc;
220  p=s*(2.0*xm*q*(q-r)-(b-a)*(r-1.0));
221  q=(q-1.0)*(r-1.0)*(s-1.0);
222  }
223  if (p > 0.0) q = -q;
224  p=std::abs(p);
225  min1=3.0*xm*q-std::abs(tol1*q);
226  min2=std::abs(e*q);
227  if (2.0*p < std::min(min1,min2) ) {
228  e=d;
229  d=p/q;
230  } else {
231  d=xm;
232  e=d;
233  }
234  } else {
235  d=xm;
236  e=d;
237  }
238  a=b;
239  fa=fb;
240  if (std::abs(d) > tol1) b += d;
241  else b += (xm>=0. ? std::abs(tol1) : -std::abs(tol1));
242  fb=func(b);
243  }
244  throw SolveError("Maximum number of iterations exceeded in zbrent");
245  }
