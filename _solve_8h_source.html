00262         // General purpose one-sided bracket with limit.  
00263         // Used by bracketUpperWithLimit and bracketLowerWithLimit.
00264         // Expand in the direction of b.
00265         bool bracket1WithLimit(T& a, T& b, T& fa, T& fb, T& c)
00266         {
00267             const T factor=2.0;
00268             // The principal here is to use z = (b-a)/(b-c) as our variable to double each time
00269             // where a is the initial lBound, c is the upper_limit, and b is the uBound we
00270             // are trying to find.
00271             //
00272             // z = (b-a)/(b-c)
00273             // b-a = zb-zc
00274             // b = (zc-a)/(z-1)
00275             //
00276             // So if we take z' = f*z
00277             // z' = f(b-a)/(b-c)
00278             // Then
00279             // b' = (z'c-a)/(z'-1)
00280             // b' = (fcb-fca-ab+ac)/(b-c) / (fb-fa-b+c)/(b-c)
00281             //    = ((fc-a)b-(f-1)ac) / ((f-1)b-fa+c)
00282             //    = (fc(b-a) + a(c-b)) / (f(b-a) + (c-b))
00283             // This is our fundamental iteration formula.
00284             //
00285             // To see the behavior of this iteration, take two limiting cases.
00286             // 1) If b = a+d and d << c-a, and use f=2 for simplicity:
00287             // b' = (2cd + ac - a^2 - ad)) / (2d+c-a-d)
00288             //    = (a(c-a) + d(2c-a)) / (c-a+d)
00289             //    = (a + d(2c-a)/(c-a)) / (1+d/(c-a))
00290             //    = (a + d(2c-a)/(c-a)) * (1-d/(c-a))
00291             //    = a + d(2c-a)/(c-a) - da/(c-a)
00292             //    = a + d(2c-2a)/(c-a)
00293             //    = a + 2d
00294             //
00295             // 2) If b = c-d and d << c-a, and use f=2 for simplicity:
00296             // b' = (2c^2-2cd-2ca+ad) / (2c-2d-2a+d)
00297             //    = (2c(c-a) - d(2c-a)) / (2(c-a)-d)
00298             //    = (c - d(2c-a)/(2c-2a)) / (1-d/(2c-2a))
00299             //    = (c - d(2c-a)/(2c-2a)) * (1+d/(2c-2a))
00300             //    = c - d(2c-a)/2(c-a) + cd/2(c-a)
00301             //    = c - d(c-a)/2(c-a)
00302             //    = c - d/2
00303             //
00304             // So when far from the limit (1), the iteration is just like the version without 
00305             // the limit.  But when the variable approaches the limit (2), it just goes half the 
00306             // remaining distance each time.
00307 
00308             xdbg<<"Bracket with limits:\n";
00309             xdbg<<"a,b,c = "<<a<<','<<b<<','<<c<<std::endl;
00310             xdbg<<"fa,fb = "<<fa<<','<<fb<<std::endl;
00311             for (int j=1; j<maxSteps; j++) {
00312                 if (fa*fb < 0.0) return true;
00313                 T bma = b-a; // Do this before overwriting a!
00314                 T cmb = c-b;
00315                 a = b;
00316                 fa = fb;
00317                 // We divide the top and bottom by (b-a)(c-b) for stability in case one of these
00318                 // is especially larger than the other.
00319                 b = (factor*c/cmb + a/bma) / (factor/cmb + 1./bma);
00320                 xdbg<<"b -> "<<b<<std::endl;
00321                 fb = func(b);
00322                 xdbg<<"fb -> "<<fb<<std::endl;
00323             }
00324             return false;
00325         }

00433                 if (std::abs(e) >= tol1 && std::abs(fa) > std::abs(fb)) {

00436                         p=2.0*xm*s;

00439                         q=fa/fc;

00442                         q=(q-1.0)*(r-1.0)*(s-1.0);
00443                     }

00445                     p=std::abs(p);

00454                     }
