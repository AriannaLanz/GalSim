154  template <typename T>
155  void applyTo(ImageView<T> data) {
156  // Above this many e's, assume Poisson distribution =Gaussian
157  static const double MAX_POISSON=1.e5;
158  // Typedef for image row iterable
159  typedef typename ImageView<T>::iterator ImIter;
160 
161  // Add the Poisson noise first:
162  if (_gain > 0.) {
163  double sigma = _gd.getSigma(); // Save this
164 
165  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
166  ImIter ee = data.rowEnd(y);
167  for (ImIter it = data.rowBegin(y); it != ee; ++it) {
168  double electrons=*it * _gain;
169  if (electrons <= 0.) continue;
170  if (electrons < MAX_POISSON) {
171  _pd.setMean(electrons);
172  *it = T(_pd() / _gain);
173  } else {
174  // ??? This might be even slower than large-N Poisson...
175  _gd.setSigma(sqrt(electrons)/_gain);
176  *it = T(*it + _gd());
177  }
178  }
179  }
180  _gd.setSigma(sigma); // restore GaussianDeviate's level
181  }
182 
183  // Next add the Gaussian noise:
184  if (_readNoise > 0.) {
185  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
186  ImIter ee = data.rowEnd(y);
187  for (ImIter it = data.rowBegin(y); it != ee; ++it) { *it = T(*it + _gd()); }
188  }
189  }
190  }

201  template <class T>
202  void applyToVar(ImageView<T> data, ImageView<T> variance) {
203  // Typedef for image row iterable
204  typedef typename ImageView<T>::iterator ImIter;
205  // Resize the variance image to match data image
206  variance.resize(data.getBounds());
207  // Fill with the (constant) Gaussian contribution to variance
208  double gaussVariance = (_readNoise > 0. ? std::pow(_gd.getSigma(),2.) : 0.);
209  variance.fill(gaussVariance);
210  // Add the Poisson variance:
211  if (_gain > 0.) {
212  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
213  ImIter ee = data.rowEnd(y);
214  ImIter it2 = variance.rowBegin(y);
215  for (ImIter it = data.rowBegin(y); it != ee; ++it, ++it2) {
216  if (*it > 0.) *it2 += *it / _gain;
217  }
218  }
219  }
220  // then call noise method to instantiate noise
221  (*this).applyTo(data);
222  }
