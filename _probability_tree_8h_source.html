247  void buildShortcut(const Element* element, int i1, int i2)
248  {
249  // If i1 == i2, then we've already assigned everything, so stop recursing.
250  if (i1 == i2) return;
251 
252  xdbg<<"Start buildShortcut for i1,i2 = "<<i1<<','<<i2<<std::endl;
253  xdbg<<"Corresponds to flux "<<i1*_totalAbsFlux/_shortcut.size()<<" .. "<<
254  i2*_totalAbsFlux/_shortcut.size()<<std::endl;
255  xdbg<<"element's flux range is "<<element->getLeftAbsFlux()<<" .. "<<
256  element->getLeftAbsFlux() + element->getAbsFlux()<<std::endl;
257  // Figure out which bins in the shortcut vector should point to this element.
258  // On input, we are tasked with assigning indices i1 <= i < i2 to be either
259  // this element or one of its decendents.
260  xassert(i1*_totalAbsFlux/_shortcut.size() >= element->getLeftAbsFlux()-1.e-8);
261  xassert(i2*_totalAbsFlux/_shortcut.size() <=
262  element->getLeftAbsFlux()+element->getAbsFlux()+1.e-8);
263 
264  // If this is a node, then the only one we should assign is the shortcut
265  // bin that include both left and right. In other words the bin corresponding
266  // to the dividing flux.
267  if (element->isNode()) {
268  double f = element->getRight()->getLeftAbsFlux();
269  int imid = int(f * _shortcut.size() / _totalAbsFlux);
270  xdbg<<"Node element: f = "<<f<<", imid = "<<imid<<std::endl;
271  if (imid < i1) {
272  // Then the appropriate range is all in the right subtree
273  buildShortcut(element->getRight(), i1, i2);
274  } else if (imid >= i2) {
275  // Then the appropriate range is all in the left subtree
276  buildShortcut(element->getLeft(), i1, i2);
277  } else {
278  // Then there is an unassigned bin that spans both children.
279  // Set it to this element.
280  assert(imid >= i1);
281  assert(imid < i2);
282  _shortcut[imid] = element;
283  // Continue on with the sub-ranges on each side.
284  buildShortcut(element->getLeft(), i1, imid);
285  buildShortcut(element->getRight(), imid+1, i2);
286  }
287  } else {
288  xdbg<<"Leaf element\n";
289  // If we are at a leaf, then this leaf encompasses all the bins in the range.
290  // Assigne them all to this element.
291  for(int i=i1; i<i2; ++i) _shortcut[i] = element;
292  }
293  }
