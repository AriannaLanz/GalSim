00123         boost::shared_ptr<Value> get(const Key& key)
00124         {
00125             assert(_entries.size() == _cache.size());
00126             MapIter iter = _cache.find(key);
00127             if (iter != _cache.end()) {
00128                 // Item is cached.
00129                 // Move it to the front of the list.
00130                 if (iter != _cache.begin()) 
00131                     _entries.splice(_entries.begin(), _entries, iter->second);
00132                 // Return the item's value
00133                 assert(_entries.size() == _cache.size());
00134                 return iter->second->second;
00135             } else {
00136                 // Item is not cached.
00137                 // Make a new one.
00138                 boost::shared_ptr<Value> value(LRUCacheHelper<Value,Key>::NewValue(key));
00139                 // Remove items from the cache as necessary.
00140                 while (_entries.size() >= _nmax) {
00141                     bool erased = _cache.erase(_entries.back().first);
00142                     assert(erased);
00143                     _entries.pop_back();
00144                 }
00145                 // Add the new value to the front.
00146                 _entries.push_front(Entry(key,value));
00147                 // Also put it in the cache
00148                 _cache[key] = _entries.begin();
00149                 // Return the new value
00150                 assert(_entries.size() == _cache.size());
00151                 return value;
00152             }
00153         }
