141  BOOST_RANDOM_DETAIL_ARITHMETIC_SEED(mersenne_twister_engine, UIntType, value)
142  {
143  // New seeding algorithm from
144  // http://www.math.sci.hiroshima-u.ac.jp/~m-mat/MT/MT2002/emt19937ar.html
145  // In the previous versions, MSBs of the seed affected only MSBs of the
146  // state x[].
147  const UIntType mask = (max)();
148  x[0] = value & mask;
149  for (i = 1; i < n; i++) {
150  // See Knuth "The Art of Computer Programming"
151  // Vol. 2, 3rd ed., page 106
152  x[i] = (f * (x[i-1] ^ (x[i-1] >> (w-2))) + i) & mask;
153  }
154  }

278  bool equal_imp(const mersenne_twister_engine& other) const
279  {
280  UIntType back[n];
281  std::size_t offset = other.i - i;
282  for(std::size_t j = 0; j + offset < n; ++j)
283  if(x[j] != other.x[j+offset])
284  return false;
285  rewind(&back[n-1], offset);
286  for(std::size_t j = 0; j < offset; ++j)
287  if(back[j + n - offset] != other.x[j])
288  return false;
289  return true;
290  }

316  void rewind(UIntType* last, std::size_t z) const
317  {
318  const UIntType upper_mask = (~static_cast<UIntType>(0)) << r;
319  const UIntType lower_mask = ~upper_mask;
320  UIntType y0 = x[m-1] ^ x[n-1];
321  if(y0 & (static_cast<UIntType>(1) << (w-1))) {
322  y0 = ((y0 ^ a) << 1) | 1;
323  } else {
324  y0 = y0 << 1;
325  }
326  for(std::size_t sz = 0; sz < z; ++sz) {
327  UIntType y1 =
328  rewind_find(last, sz, m-1) ^ rewind_find(last, sz, n-1);
329  if(y1 & (static_cast<UIntType>(1) << (w-1))) {
330  y1 = ((y1 ^ a) << 1) | 1;
331  } else {
332  y1 = y1 << 1;
333  }
334  *(last - sz) = (y0 & upper_mask) | (y1 & lower_mask);
335  y0 = y1;
336  }
337  }

344  UIntType
345  rewind_find(UIntType* last, std::size_t size, std::size_t j) const
346  {
347  std::size_t index = (j + n - size + n - 1) % n;
348  if(index < n - size) {
349  return x[index];
350  } else {
351  return *(last - (n - 1 - index));
352  }

401  UIntType a, std::size_t u, UIntType d, std::size_t s,

404 void
405 mersenne_twister_engine<UIntType,w,n,m,r,a,u,d,s,b,t,c,l,f>::twist()
406 {
407  const UIntType upper_mask = (~static_cast<UIntType>(0)) << r;
408  const UIntType lower_mask = ~upper_mask;
409 
410  const std::size_t unroll_factor = 6;
411  const std::size_t unroll_extra1 = (n-m) % unroll_factor;
412  const std::size_t unroll_extra2 = (m-1) % unroll_factor;
413 
414  // split loop to avoid costly modulo operations
415  { // extra scope for MSVC brokenness w.r.t. for scope
416  for(std::size_t j = 0; j < n-m-unroll_extra1; j++) {
417  UIntType y = (x[j] & upper_mask) | (x[j+1] & lower_mask);
418  x[j] = x[j+m] ^ (y >> 1) ^ ((x[j+1]&1) * a);
419  }
420  }
421  {
422  for(std::size_t j = n-m-unroll_extra1; j < n-m; j++) {
423  UIntType y = (x[j] & upper_mask) | (x[j+1] & lower_mask);
424  x[j] = x[j+m] ^ (y >> 1) ^ ((x[j+1]&1) * a);
425  }
426  }
427  {
428  for(std::size_t j = n-m; j < n-1-unroll_extra2; j++) {
429  UIntType y = (x[j] & upper_mask) | (x[j+1] & lower_mask);
430  x[j] = x[j-(n-m)] ^ (y >> 1) ^ ((x[j+1]&1) * a);
431  }
432  }
433  {
434  for(std::size_t j = n-1-unroll_extra2; j < n-1; j++) {
435  UIntType y = (x[j] & upper_mask) | (x[j+1] & lower_mask);
436  x[j] = x[j-(n-m)] ^ (y >> 1) ^ ((x[j+1]&1) * a);
437  }
438  }
439  // last iteration
440  UIntType y = (x[n-1] & upper_mask) | (x[0] & lower_mask);
441  x[n-1] = x[m-1] ^ (y >> 1) ^ ((x[0]&1) * a);
442  i = 0;
443 }

447  std::size_t w, std::size_t n, std::size_t m, std::size_t r,
448  UIntType a, std::size_t u, UIntType d, std::size_t s,

451 inline typename
452 mersenne_twister_engine<UIntType,w,n,m,r,a,u,d,s,b,t,c,l,f>::result_type
453 mersenne_twister_engine<UIntType,w,n,m,r,a,u,d,s,b,t,c,l,f>::operator()()
454 {
455  if(i == n)
456  twist();
457  // Step 4
458  UIntType z = x[i];
459  ++i;
460  z ^= ((z >> u) & d);
461  z ^= ((z << s) & b);
462  z ^= ((z << t) & c);
463  z ^= (z >> l);
464  return z;
465 }
