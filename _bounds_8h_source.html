361  template <class T>
362  std::vector< Bounds<T> > Bounds<T>::divide(int nx, int ny) const
363  {
364  if (!defined) return std::vector< Bounds<T> >(nx,ny);
365  typename std::vector< Bounds<T> > temp(nx*ny);
366  std::vector<double> x(nx+1);
367  std::vector<double> y(ny+1);
368  x[0] = xmin; x[nx] = xmax;
369  y[0] = ymin; y[ny] = ymax;
370  double xstep = (xmax-xmin)/nx;
371  double ystep = (ymax-ymin)/ny;
372  for(int i=1;i<nx;i++) x[i] = x[0]+i*xstep;
373  for(int j=1;j<ny;j++) y[j] = y[0]+j*ystep;
374  typename std::vector< Bounds<T> >::iterator ii=temp.begin();
375  for(int i=0;i<nx;i++) for(int j=0;j<ny;j++, ++i)
376  *ii = Bounds<T>(x[i],x[i+1],y[j],y[j+1]);
377  return temp;
378  }
