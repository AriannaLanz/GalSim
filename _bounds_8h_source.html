319  template <class T>
320  std::vector< Bounds<T> > Bounds<T>::divide(int nx, int ny) const
321  {
322  if (!defined) return std::vector< Bounds<T> >(nx,ny);
323  typename std::vector< Bounds<T> > temp(nx*ny);
324  std::vector<double> x(nx+1);
325  std::vector<double> y(ny+1);
326  x[0] = xmin; x[nx] = xmax;
327  y[0] = ymin; y[ny] = ymax;
328  double xstep = (xmax-xmin)/nx;
329  double ystep = (ymax-ymin)/ny;
330  for(int i=1;i<nx;i++) x[i] = x[0]+i*xstep;
331  for(int j=1;j<ny;j++) y[j] = y[0]+j*ystep;
332  typename std::vector< Bounds<T> >::iterator ii=temp.begin();
333  for(int i=0;i<nx;i++) for(int j=0;j<ny;j++, ++i)
334  *ii = Bounds<T>(x[i],x[i+1],y[j],y[j+1]);
335  return temp;
