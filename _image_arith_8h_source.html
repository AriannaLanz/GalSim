00068     template <typename T, typename Op>
00069     Op for_each_pixel(const ImageView<T>& image, Op f) 
00070     {
00071         // Note: all of these functions have this guard to make sure we don't
00072         // try to access the memory if the image is in an undefined state.
00073         if (image.getData()) {
00074             if (image.isContiguous()) {
00075                 f = std::for_each(image.rowBegin(image.getYMin()), image.rowEnd(image.getYMax()), f);
00076             } else {
00077                 for (int i = image.getYMin(); i <= image.getYMax(); i++)
00078                     f = std::for_each(image.rowBegin(i), image.rowEnd(i), f);
00079             }
00080         }
00081         return f;
00082     }

00087     template <typename T, typename Op>
00088     Op for_each_pixel(const ImageView<T>& image, const Bounds<int>& bounds, Op f) 
00089     {
00090         if (image.getData()) {
00091             if (!image.getBounds().includes(bounds))
00092                 throw ImageError("for_each_pixel range exceeds image range");
00093 
00094             if (image.getBounds() == bounds) return for_each_pixel(image,f);
00095 
00096             for (int i = bounds.getYMin(); i <= bounds.getYMax(); i++)
00097                 f = std::for_each(image.getIter(bounds.getXMin(),i),
00098                                   image.getIter(bounds.getXMax()+1,i), f);
00099         }
00100         return f;
00101     }

00232     template <typename T1, typename T2, typename Op>
00233     Op transform_pixel(const ImageView<T1>& image1, const BaseImage<T2>& image2, Op f) 
00234     {
00235         if (image1.getData()) {
00236             typedef typename ImageView<T1>::iterator Iter1;
00237             typedef typename BaseImage<T2>::const_iterator Iter2;
00238 
00239             if (!image1.getBounds().isSameShapeAs(image2.getBounds()))
00240                 throw ImageError("transform_pixel image bounds are not same shape");
00241 
00242             int y2 = image2.getYMin();
00243             for (int y = image1.getYMin(); y <= image1.getYMax(); ++y, ++y2) {
00244                 Iter2 it2 = image2.rowBegin(y2);
00245                 const Iter1 ee = image1.rowEnd(y);      
00246                 for (Iter1 it1 = image1.rowBegin(y); it1 != ee; ++it1, ++it2)
00247                     *it1 = f(*it1,*it2);
00248             }
00249         }
00250         return f;
00251     }

00269                 throw ImageError("transform_pixel image1, image3 bounds are not same shape");
00270

00272             int y3 = image3.getYMin();
00273             for (int y = image1.getYMin(); y <= image1.getYMax(); ++y, ++y2, ++y3) {
00274                 Iter2 it2 = image2.rowBegin(y2);
00275                 Iter3 it3 = image3.rowBegin(y3);
00276                 const Iter1 ee = image1.rowEnd(y);      
00277                 for (Iter1 it1 = image1.rowBegin(y); it1 != ee; ++it1, ++it2, ++it3) 
00278                     *it1 = f(*it2,*it3);
