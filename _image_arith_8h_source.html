266  Op transform_pixel(
267  const ImageView<T1>& image1,
268  const BaseImage<T2>& image2,
269  const Bounds<int>& bounds,
270  Op f)
271  {
272  if (image1.getData()) {
273  typedef typename ImageView<T1>::iterator Iter1;
