00552         {
00553             if (!this->_bounds.isSameShapeAs(rhs.getBounds()))
00554                 throw ImageError("Attempt im1 = im2, but bounds not the same shape");
00555             for (int y=this->getYMin(), y2=rhs.getYMin(); y <= this->getYMax(); ++y, ++y2) {
00556                 iterator it1 = rowBegin(y);
00557                 const iterator ee = rowEnd(y);      
00558                 typename BaseImage<U>::const_iterator it2 = rhs.rowBegin(y2);
00559                 while (it1 != ee) *(it1++) = T(*(it2++));
