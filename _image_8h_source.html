602  template <class U>
603  void copyFrom(const BaseImage<U>& rhs) const
604  {
605  if (!this->getBounds().isSameShapeAs(rhs.getBounds()))
606  throw ImageError("Attempt im1 = im2, but bounds not the same shape");
607  for (int y=this->getYMin(), y2=rhs.getYMin(); y <= this->getYMax(); ++y, ++y2) {
608  iterator it1 = rowBegin(y);
609  const iterator ee = rowEnd(y);
610  typename BaseImage<U>::const_iterator it2 = rhs.rowBegin(y2);
611  while (it1 != ee) *(it1++) = T(*(it2++));
612  }
613  }
