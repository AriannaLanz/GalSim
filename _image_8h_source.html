589  template <class U>
590  void copyFrom(const BaseImage<U>& rhs) const
591  {
592  if (!this->getBounds().isSameShapeAs(rhs.getBounds()))
593  throw ImageError("Attempt im1 = im2, but bounds not the same shape");
594  for (int y=this->getYMin(), y2=rhs.getYMin(); y <= this->getYMax(); ++y, ++y2) {
595  iterator it1 = rowBegin(y);
596  const iterator ee = rowEnd(y);
597  typename BaseImage<U>::const_iterator it2 = rhs.rowBegin(y2);
598  while (it1 != ee) *(it1++) = *(it2++);
599  }
600  }
