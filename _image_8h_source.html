550  template <class U>
551  void copyFrom(const BaseImage<U>& rhs) const
552  {
553  if (!this->_bounds.isSameShapeAs(rhs.getBounds()))
554  throw ImageError("Attempt im1 = im2, but bounds not the same shape");
555  for (int y=this->getYMin(), y2=rhs.getYMin(); y <= this->getYMax(); ++y, ++y2) {
556  iterator it1 = rowBegin(y);
557  const iterator ee = rowEnd(y);
558  typename BaseImage<U>::const_iterator it2 = rhs.rowBegin(y2);
559  while (it1 != ee) *(it1++) = T(*(it2++));
560  }
561  }
