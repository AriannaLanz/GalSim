569  template <class U>
570  void copyFrom(const BaseImage<U>& rhs) const
571  {
572  if (!this->getBounds().isSameShapeAs(rhs.getBounds()))
573  throw ImageError("Attempt im1 = im2, but bounds not the same shape");
574  for (int y=this->getYMin(), y2=rhs.getYMin(); y <= this->getYMax(); ++y, ++y2) {
575  iterator it1 = rowBegin(y);
576  const iterator ee = rowEnd(y);
577  typename BaseImage<U>::const_iterator it2 = rhs.rowBegin(y2);
578  while (it1 != ee) *(it1++) = *(it2++);
579  }
