83  // Function for applying deviates to an image... Used as a method for all Deviates below.
84  template <typename D, typename T>
85  static void ApplyDeviateToImage(D& dev, ImageView<T>& data)
86  {
87  // Typedef for image row iterable
88  typedef typename ImageView<T>::iterator ImIter;
89 
90  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
91  ImIter ee = data.rowEnd(y);
92  for (ImIter it = data.rowBegin(y); it != ee; ++it) { *it += T(dev()); }
93  }
94  }
