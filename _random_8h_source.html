45  // Function for applying deviates to an image... Used as a method for all Deviates below.
46  template <typename D, typename T>
47  static void ApplyDeviateToImage(D& dev, ImageView<T>& data)
48  {
49  // Typedef for image row iterable
50  typedef typename ImageView<T>::iterator ImIter;
51 
52  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
53  ImIter ee = data.rowEnd(y);
54  for (ImIter it = data.rowBegin(y); it != ee; ++it) { *it += dev(); }
55  }
56  }
