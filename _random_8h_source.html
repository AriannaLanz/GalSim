82  // Function for applying deviates to an image... Used as a method for all Deviates below.
83  template <typename D, typename T>
84  static void ApplyDeviateToImage(D& dev, ImageView<T>& data)
85  {
86  // Typedef for image row iterable
87  typedef typename ImageView<T>::iterator ImIter;
88 
89  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
90  ImIter ee = data.rowEnd(y);
91  for (ImIter it = data.rowBegin(y); it != ee; ++it) { *it += T(dev()); }
92  }
93  }
