78  // Function for applying deviates to an image... Used as a method for all Deviates below.
79  template <typename D, typename T>
80  static void ApplyDeviateToImage(D& dev, ImageView<T>& data)
81  {
82  // Typedef for image row iterable
83  typedef typename ImageView<T>::iterator ImIter;
84 
85  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
86  ImIter ee = data.rowEnd(y);
87  for (ImIter it = data.rowBegin(y); it != ee; ++it) { *it += T(dev()); }
88  }
89  }
