58  // Function for applying deviates to an image... Used as a method for all Deviates below.
59  template <typename D, typename T>
60  static void ApplyDeviateToImage(D& dev, ImageView<T>& data)
61  {
62  // Typedef for image row iterable
63  typedef typename ImageView<T>::iterator ImIter;
64 
65  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
66  ImIter ee = data.rowEnd(y);
67  for (ImIter it = data.rowBegin(y); it != ee; ++it) { *it += T(dev()); }
68  }
69  }
