00082     // Function for applying deviates to an image... Used as a method for all Deviates below.
00083     template <typename D, typename T>
00084     static void ApplyDeviateToImage(D& dev, ImageView<T>& data) 
00085     {
00086         // Typedef for image row iterable
00087         typedef typename ImageView<T>::iterator ImIter;
00088 
00089         for (int y = data.getYMin(); y <= data.getYMax(); y++) {  // iterate over y
00090             ImIter ee = data.rowEnd(y);
00091             for (ImIter it = data.rowBegin(y); it != ee; ++it) { *it += T(dev()); }
00092         }
00093     }
