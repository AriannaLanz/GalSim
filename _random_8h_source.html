81  // Function for applying deviates to an image... Used as a method for all Deviates below.
82  template <typename D, typename T>
83  static void ApplyDeviateToImage(D& dev, ImageView<T>& data)
84  {
85  // Typedef for image row iterable
86  typedef typename ImageView<T>::iterator ImIter;
87 
88  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
89  ImIter ee = data.rowEnd(y);
90  for (ImIter it = data.rowBegin(y); it != ee; ++it) { *it += T(dev()); }
91  }
92  }
